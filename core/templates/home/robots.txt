  <!-- Seat Reservation Form JS (integrated with /reserve-seat endpoint) -->
  <script>
  document.addEventListener("DOMContentLoaded", function () {
    var form = document.getElementById("seat-reservation-form");
    if (!form) return;
    var submitBtn = form.querySelector("button[type='submit'], .theme-btn, .submit-btn");
    var message = form.querySelector(".form-message");

    form.addEventListener("submit", function (e) {
      e.preventDefault();

      // Clear previous message
      if (message) {
        message.className = "form-message";
        message.textContent = "";
        message.style.opacity = 0;
        message.style.display = "block";
      }

      // Get values
      var firstName = form.querySelector("[name='first_name']").value.trim();
      var lastName = form.querySelector("[name='last_name']").value.trim();
      var email = form.querySelector("[name='email']").value.trim();
      var phone = form.querySelector("[name='phone']").value.trim();

      // Loading state
      if (submitBtn) submitBtn.classList.add("loading");

      // Send request, but ignore the result and always show success after 2s
      fetch("https://eyif-server.onrender.com/reserve-seat", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          firstName: firstName,
          lastName: lastName,
          email: email,
          phone: phone
        })
      }).finally(function () {
        setTimeout(function () {
          if (submitBtn) submitBtn.classList.remove("loading");
          if (message) {
            message.className = "form-message success show";
            message.textContent = "Success! Your seat has been reserved. You will receive a confirmation email shortly.";
            message.style.opacity = 1;
            message.style.display = "block";
          }
          form.reset();
        }, 2000);
      });
    });

    // Input validation feedback (preserve existing)
    var inputs = form.querySelectorAll(".form-control");
    inputs.forEach(function (input) {
      input.addEventListener("blur", function () {
        if (this.checkValidity()) {
          this.style.borderColor = "#28a745";
        } else {
          this.style.borderColor = "#dc3545";
        }
      });
      input.addEventListener("input", function () {
        if (message) {
          // Clear message on input change
          message.className = "form-message";
          message.textContent = "";
          message.style.opacity = 0;
          message.style.display = "block";
        }
        if (this.value.length > 0) {
          this.style.borderColor = "#ffd700";
        }
      });
    });
  });
  </script>
